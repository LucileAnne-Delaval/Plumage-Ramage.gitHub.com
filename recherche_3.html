float angle = 0;
import ddf.minim.*;
import ddf.minim.analysis.*;
Minim minim;
AudioPlayer song;
BeatDetect beat;
float eRadius;
float a = 0;

void setup() {
size(600, 600, P3D);
noStroke();
minim = new Minim(this);
song = minim.loadFile("martin-pecheur_005.mp3", 2048);
song.play();
beat = new BeatDetect();
eRadius = 10; }

void draw() {
beat.detect(song.mix);
float a = map(eRadius, 1, 10, 20, 3);
if ( beat.isOnset() ) eRadius = 80;
pushMatrix();
translate(width/2, height/2);
rotateY(map(a , 0, width, 0, PI));
rotateX(map(a, 0, height, 0, PI));
noFill();
stroke (230, 127, 0, 3);
box(300);
popMatrix();
pushMatrix();
translate(width/2, height/2);
rotateY(map(-a , 0, width, 0, PI));
rotateX(map(-a, 0, height, 0, PI));
noFill();
stroke (0, 102, 255, 3);
box(400);
popMatrix();
eRadius *= 0.95;
if ( eRadius < 20 ) eRadius -=30;
}
