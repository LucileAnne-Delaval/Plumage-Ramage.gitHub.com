float angle = 0;
import ddf.minim.*;
import ddf.minim.analysis.*;
Minim minim;
AudioPlayer song;
BeatDetect beat;
float eRadius;
float a = 0;

void setup() {
size(600, 600, P3D);
noStroke();
minim = new Minim(this);
song = minim.loadFile("martin-pecheur_005.mp3", 2048);
song.play();
beat = new BeatDetect();
eRadius = 10; }

void draw() {
defineLights();
background(0);
beat.detect(song.mix);
float a = map(eRadius, 20, 80, 60, 255);

if ( beat.isOnset() ) eRadius = 80;
for (int x = 0; x <= width; x += 60) {
for (int y = 0; y <= height; y += 60) {
pushMatrix();
translate(x, y);
rotateY(map(a , 0, width, 0, PI));
rotateX(map(a, 0, height, 0, PI));
box(80);
popMatrix(); } }
eRadius *= 0.95;
if ( eRadius < 20 ) eRadius -=30;}

void defineLights() {
// Orange point light on the right
pointLight(230, 127, 0, 100, -50, 0);

// Blue directional light from the left
directionalLight(0, 102, 255, 100, 200, 0);

// Yellow spotlight from the front
spotLight(255, 155, 109, 0, 40, 200, 0, -0.5, -0.5, PI / 2, 2); }
