import ddf.minim.*;
Minim minim;
AudioPlayer player;
boolean a = false ;
int x = 0;
int y= 255;

void setup () {
size (600, 600);
minim = new Minim(this);
player = minim.loadFile("attirer.mp3");
player.play();
background (0); }

void draw () {
scale (0.5);
noStroke();
float d = map (player.mix.level(), 0, 0.5, 50, 250);
float c = map (player.mix.level(), 0, 0.5, 0, 500);
fill(255);
if (x<360) {
translate (width,height);
rotate (radians(x));
ellipse (c, c, d, d);
x+=0.5; }
}
